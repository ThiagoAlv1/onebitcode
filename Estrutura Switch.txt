Exemplo:

const resultado = prompt("Escolha uma alternativa A, B, C")

switch (resultado) {
    case "a": 
    alert("O resultado é A")
    break
    
    case "b":
    alert("O resultado é B")
    break

    case "c":
    alert("O resultado é C")
    break
    
    default:
    alert("Finalizando...")
}
